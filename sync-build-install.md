## Sync fork with upstream

```bash

cd ~/develop/zed

# One-time setup:
# git remote add upstream https://github.com/zed-industries/zed.git

git fetch upstream
git checkout main
git merge upstream/main

# Resolve conflicts if needed, then finish the merge.
git commit

git push origin main
```

## Build and install two local apps

Use one app bundle for your clean `main` build and another for your testing branch.
The binary always comes from the branch you just built.

```bash


REPO_ROOT="$HOME/develop/zed"
OFFICIAL_APP="/Applications/Zed.app"
MAIN_APP="/Applications/Zed Main.app"
DEV_APP="/Applications/Zed Dev.app"

find_bundled_dev_icon() {
    find "$REPO_ROOT/target" \
        -path "*/bundle/osx/Zed Dev.app/Contents/Resources/Zed Dev.icns" \
        -print \
        -quit 2>/dev/null
}

install_zed_variant() {
    local variant="$1"
    local app_path app_name bundle_id icon_file icon_source

    case "$variant" in
        main)
            app_path="$MAIN_APP"
            app_name="Zed Main"
            bundle_id="dev.zed.Zed-Main"
            icon_file="Zed.icns"
            icon_source="$OFFICIAL_APP/Contents/Resources/Zed.icns"
            ;;
        dev)
            app_path="$DEV_APP"
            app_name="Zed Dev"
            bundle_id="dev.zed.Zed-Dev-Local"
            icon_file="Zed Dev.icns"
            icon_source="$(find_bundled_dev_icon)"
            ;;
        *)
            echo "usage: install_zed_variant main|dev"
            return 1
            ;;
    esac

    local binary="$REPO_ROOT/target/release/zed"
    local plist="$app_path/Contents/Info.plist"
    local dst_resources="$app_path/Contents/Resources"

    if [[ ! -x "$binary" ]]; then
        echo "==> ERROR: $binary not found."
        echo "==> Run: cargo build --release"
        return 1
    fi

    if [[ ! -d "$OFFICIAL_APP" ]]; then
        echo "==> ERROR: $OFFICIAL_APP not found."
        echo "==> Install the official app once so it can be used as the template bundle."
        return 1
    fi

    if [[ "$variant" == "dev" && -z "$icon_source" ]]; then
        echo "==> ERROR: no bundled dev icon found under target/*/bundle/osx."
        echo "==> Run: script/bundle-mac"
        echo "==> Then rerun: install_zed_variant dev"
        return 1
    fi

    if [[ ! -d "$app_path" ]]; then
        echo "==> Creating $app_path from $OFFICIAL_APP"
        ditto "$OFFICIAL_APP" "$app_path"
    fi

    echo "==> Installing $variant build into $app_path"
    cp "$binary" "$app_path/Contents/MacOS/zed"

    echo "==> Updating app metadata"
    /usr/libexec/PlistBuddy -c "Set :CFBundleDisplayName $app_name" "$plist" || \
        /usr/libexec/PlistBuddy -c "Add :CFBundleDisplayName string $app_name" "$plist"
    /usr/libexec/PlistBuddy -c "Set :CFBundleName $app_name" "$plist" || \
        /usr/libexec/PlistBuddy -c "Add :CFBundleName string $app_name" "$plist"
    /usr/libexec/PlistBuddy -c "Set :CFBundleIdentifier $bundle_id" "$plist"
    /usr/libexec/PlistBuddy -c "Set :CFBundleIconFile $icon_file" "$plist" || \
        /usr/libexec/PlistBuddy -c "Add :CFBundleIconFile string $icon_file" "$plist"

    echo "==> Refreshing icons"
    cp "$icon_source" "$dst_resources/$icon_file"
    cp "$OFFICIAL_APP/Contents/Resources/Document.icns" "$dst_resources/Document.icns"

    touch "$app_path"
    /usr/bin/killall Dock 2>/dev/null || true

    echo "==> Done: $app_path"
}
```

## Typical flow

### Build and install the `main` app

```bash
git checkout main
cargo build --release
install_zed_variant main
```

### Build and install the testing app

```bash
git checkout send-code-v2
cargo build --release
install_zed_variant dev
```

## Notes

- `main` installs into `/Applications/Zed Main.app`.
- `dev` installs into `/Applications/Zed Dev.app`.
- `dev` uses the bundled `Zed Dev.icns` generated by `cargo bundle` / `script/bundle-mac`.
- The first install clones `/Applications/Zed.app` as the template bundle.
- You must relaunch the app after replacing the binary.
